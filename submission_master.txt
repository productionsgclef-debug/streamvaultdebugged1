1. ANDROID: build.gradle (Module: app)
=====================================================================
# Add this inside the 'android' block to ensure 64-bit compliance 
# and Aptoide compatibility.

android {
    defaultConfig {
        // Essential for Google Play & Aptoide 64-bit requirements
        ndk {
            abiFilters 'armeabi-v7a', 'arm64-v8a', 'x86', 'x86_64'
        }
    }

    # Enable App Bundles for Google Play delivery
    bundle {
        abi { enableSplit = true }
        density { enableSplit = true }
        language { enableSplit = true }
    }
}

3. APTOIDE: Universal APK Conversion Script
=====================================================================
# Run this command using bundletool to generate the APK Aptoide needs.
# Replace placeholders with your actual keystore details.

java -jar bundletool.jar build-apks \
--bundle=YOUR_APP.aab \
--output=YOUR_APP.apks \
--mode=universal \
--ks=YOUR_KEYSTORE.jks \
--ks-pass=pass:YOUR_PASSWORD \
--ks-key-alias=YOUR_ALIAS \
--key-pass=pass:YOUR_PASSWORD

# After running:
# 1. Rename YOUR_APP.apks to YOUR_APP.zip
# 2. Extract the file.
# 3. Upload 'universal.apk' to Aptoide.
